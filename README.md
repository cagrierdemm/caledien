## Install

```bash
npm install --save caledien
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'caledien'
import 'caledien/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
